function doGet(e) { var prop = PropertiesService.getScriptProperties(); var callback = e.parameter.callback; var action = e.parameter.action; if (action === 'save') { prop.setProperty('data', e.parameter.payload); var res = JSON.stringify({status:'ok'}); if (callback) return ContentService.createTextOutput(callback+'('+res+')').setMimeType(ContentService.MimeType.JAVASCRIPT); return ContentService.createTextOutput(res).setMimeType(ContentService.MimeType.JSON); } if (action === 'saveWeek') { var raw = prop.getProperty('data') || '{}'; var obj = {}; try { obj = JSON.parse(raw); } catch(err) {} obj.lastViewedWeek = e.parameter.week; prop.setProperty('data', JSON.stringify(obj)); var res2 = JSON.stringify({status:'ok'}); if (callback) return ContentService.createTextOutput(callback+'('+res2+')').setMimeType(ContentService.MimeType.JAVASCRIPT); return ContentService.createTextOutput(res2).setMimeType(ContentService.MimeType.JSON); } var data = prop.getProperty('data') || '{}'; if (callback) return ContentService.createTextOutput(callback+'('+data+')').setMimeType(ContentService.MimeType.JAVASCRIPT); return ContentService.createTextOutput(data).setMimeType(ContentService.MimeType.JSON); }
